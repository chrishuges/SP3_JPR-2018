Processing of data comparing different SP3 solvent conditions

Description

This file contains the code for processing the search and quantification output files from the examination of the performance of SP3 in different solvent conditions with in-depth proteome analysis. These data were used to make portions of Figure XXXXX.

  • MGF files were generated using RawQuant.
  • The data were searched using SearchCLI and PeptideShakerCLI.
  • Raw files were parsed for quantification data with RawQuant.

    dplyr::mutate(Spectrum.Title, scan_num = sub(".*?spectrum (.*?)", "\\1", Spectrum.Title)) %>% #parse out the scan number
    dplyr::mutate(Identification.Charge, id_charge = gsub("[^0-9]", "", Identification.Charge)) %>% #parse out the charge
    dplyr::mutate(Spectrum.Title, fraction = sub(".*?hph_(.*?)_MGF\\.mgf", "\\1", Spectrum.File)) %>% #parse the fraction
    dplyr::select(Protein.s., Sequence, scan_num, m.z, id_charge, fraction) %>% #select new columns
    dplyr::mutate(Protein.s., first_accession = sapply(strsplit(as.character(Protein.s.),"\\,"), `[`, 1)) #parse the accession
  #adjust the column names
  colnames(psm_f1) = c('accession_groups','sequence','ScanNumber','id_precursor_mz','id_charge','fraction','accession')
  psm_f2 = dplyr::filter(psm_f1, !grepl('CONT_', accession_groups)) #remove the contaminant proteins
  id_parsed = rbind(id_parsed,psm_f2) #group the data with the previously processed set
}

#calculate differential expression using limma
eset = as.matrix(log2(pro_set[,c(3:8)])) #log transform the expression data and transfer to a new object
design <- cbind(HIGH=c(1,1,1,0,0,0),LOW=c(0,0,0,1,1,1)) #create the design matrix
fit <- lmFit(eset,design) #calculate the linear model of the expression values
cont.matrix <- makeContrasts(HIGHvsLOW=HIGH-LOW, levels=design) #create the contrasts matrix
fit2 <- contrasts.fit(fit, cont.matrix) #calculate the contrasts
fit2 <- eBayes(fit2) #empirical Bayes smoothing
logFC = fit2$coef #extract fold changes
pVal = fit2$p.value #extract p-values
p.adj = p.adjust(fit2$p.value, method="BH") #apply multiple testing correction to the p-values
#insert the data back into the original matrix
pro_set$logFC <- logFC
pro_set$p.value <- pVal
pro_set$p.adj.value <- p.adj
pro_ordered = pro_set[order(-pro_set$logFC),] #reorder the matrix by fold change
